```python
def _desenhar_painel_modal(
    surface: pygame.Surface,
    painel: pygame.Rect,
    titulo: str,
    fonte_titulo: pygame.font.Font,
) -> None:
    """Desenha fundo, bordas e header no estilo .modal-painel do HTML."""
    # Fundo
    pygame.draw.rect(surface, COR_FUNDO_MODAL, painel)
    # Borda lateral 1px
    pygame.draw.rect(surface, COR_BORDA_MODAL, painel, 1)
    # Borda superior 3px dourada
    pygame.draw.line(
        surface, COR_BORDA_MODAL_TOPO,
        painel.topleft, (painel.right, painel.top), 3
    )
    # Header: linha divisória a 50px do topo
    header_y = painel.y + 50
    pygame.draw.line(
        surface, COR_HUD_BORDA,
        (painel.x, header_y), (painel.right, header_y), 1
    )
    # Gradiente lateral sutil (quadrado dourado 4% alpha no canto esq)
    grad = pygame.Surface((painel.width // 2, 50), pygame.SRCALPHA)
    grad.fill((255, 208, 64, 10))
    surface.blit(grad, painel.topleft)
    # Título no header
    txt = fonte_titulo.render(titulo.upper(), True, COR_DOURADO)
    surface.blit(txt, (painel.x + 20, painel.y + 15))
```

```python
"""Preferências locais do jogador (persistidas em JSON).

Formato: ~/.config/speedvslabubu/config.json
{
    "dialogo_habilitado": true,
    "volume": 0.7
}
Qualquer falha de I/O é tolerada — o jogo usa os defaults.
"""

import json
import logging
import sys
from pathlib import Path

logger = logging.getLogger(__name__)

DEFAULTS: dict = {
    "dialogo_habilitado": True,
    "volume": 0.7,
}


def _get_path() -> Path:
    if sys.platform == "win32":
        base = Path.home() / "AppData" / "Roaming" / "speedvslabubu"
    else:
        base = Path.home() / ".config" / "speedvslabubu"
    base.mkdir(parents=True, exist_ok=True)
    return base / "config.json"


def carregar() -> dict:
    """Lê config.json; retorna defaults em qualquer erro."""
    try:
        path = _get_path()
        if not path.exists():
            return dict(DEFAULTS)
        dados = json.loads(path.read_text(encoding="utf-8"))
        return {k: dados.get(k, v) for k, v in DEFAULTS.items()}
    except Exception as e:  # noqa: BLE001
        logger.warning("[Preferências] Não foi possível carregar: %s", e)
        return dict(DEFAULTS)


def salvar(dados: dict) -> None:
    """Grava config.json; falha de I/O é apenas logada."""
    try:
        atual = carregar()
        atual.update(dados)
        _get_path().write_text(
            json.dumps(atual, ensure_ascii=False, indent=2), encoding="utf-8"
        )
    except Exception as e:  # noqa: BLE001
        logger.warning("[Preferências] Não foi possível salvar: %s", e)


def get(chave: str):
    """Retorna o valor de uma preferência (ou default se ausente)."""
    return carregar().get(chave, DEFAULTS.get(chave))


def set(chave: str, valor) -> None:
    """Salva uma preferência isolada."""
    salvar({chave: valor})
```
