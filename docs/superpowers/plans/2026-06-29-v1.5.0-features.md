```python
"""Sistema de prioridade de alvo para torres.

Cada torre carrega seu próprio `priority: TargetPriority`. A função
`select_target()` encapsula toda a lógica de seleção — nenhuma torre
duplica essa lógica internamente.
"""

import logging
from enum import Enum

logger = logging.getLogger(__name__)


class TargetPriority(Enum):
    """Critérios de seleção de alvo. ONLY é reservado para uso futuro."""

    FIRST    = "primeiro"     # inimigo mais avançado no path — padrão
    LAST     = "ultimo"       # inimigo mais atrás no path
    STRONGEST = "mais_forte"  # maior HP atual
    WEAKEST  = "mais_fraco"   # menor HP atual
    ONLY     = "apenas"       # uso futuro (reservado)


def select_target(
    enemies: list,
    priority: TargetPriority,
    tower_range: float,
    tower_pos: tuple[float, float],
) -> object | None:
    """Retorna o inimigo alvo conforme prioridade, ou None se nenhum no alcance.

    Parâmetros:
        enemies: lista de Enemy (pode ser vazia — retorna None sem exceção)
        priority: critério de seleção
        tower_range: alcance em pixels
        tower_pos: (x, y) do centro da torre em pixels
    """
    if not enemies:
        return None

    cx, cy = tower_pos
    alcance2 = tower_range * tower_range

    no_alcance = [
        e for e in enemies
        if (e.x - cx) ** 2 + (e.y - cy) ** 2 <= alcance2
    ]
    if not no_alcance:
        return None

    if priority == TargetPriority.FIRST:
        # Mais avançado: maior waypoint_index; empate = mais próximo do waypoint
        return max(
            no_alcance,
            key=lambda e: (e.waypoint_index, -((e.x - cx) ** 2 + (e.y - cy) ** 2)),
        )

    if priority == TargetPriority.LAST:
        return min(
            no_alcance,
            key=lambda e: (e.waypoint_index, (e.x - cx) ** 2 + (e.y - cy) ** 2),
        )

    if priority == TargetPriority.STRONGEST:
        return max(no_alcance, key=lambda e: e.hp)

    if priority == TargetPriority.WEAKEST:
        return min(no_alcance, key=lambda e: e.hp)

    # ONLY: reservado — fallback para FIRST com aviso
    logger.warning(
        "Prioridade ONLY ainda não implementada — usando FIRST como fallback."
    )
    return select_target(enemies, TargetPriority.FIRST, tower_range, tower_pos)
```

"""DrivingCar Speed — torre buffer de aceleração de ataque global.

Tipo: Buffer (não causa dano, não atira).
Efeito: Aumenta fire_rate de todas as torres ativas em 120% enquanto estiver
no campo. Stacking desativado: segunda instância não reaplica o buff.
Alcance: 85 base → 100 no upgrade (apenas para exibição — sem detecção de alvo).
Raridade: Limitado (obtida via gacha com pity em 120 pulls).
"""

import logging

from entities.tower import Tower

logger = logging.getLogger(__name__)

# Multiplicador de fire_rate aplicado globalmente.
BUFF_FIRE_RATE_MULT: float = 2.20   # +120% = fator 2.2×
# Alcance base e máximo (apenas exibição, sem uso em combate).
RANGE_BASE: int = 85
RANGE_MAX: int = 100


class DrivingCarSpeed(Tower):
    """Torre buffer: acelera fire_rate global em 120%. Não atira."""

    nome = "DrivingCar Speed"
    cost = 300
    damage = 0
    range_px = RANGE_BASE
    fire_rate = 0.0   # não atira — update() retorna None diretamente
    slow = False
    asset_name = "speed9"
    description = "Buff global: +120% cadência. Não acumula."
    max_no_campo = 1  # só uma no campo (stacking desativado)

    def __init__(self, assets, x: float, y: float, cell_x: int, cell_y: int) -> None:
        super().__init__(assets, x, y, cell_x, cell_y)
        self.buff_aplicado: bool = False  # True quando o buff global está ativo

    def update(self, dt: float, enemies: list) -> None:
        """Não atira — retorna None sem entrar na lógica de fire_rate."""
        return None

    def on_place(self, towers: list["Tower"]) -> None:
        """Aplica buff de fire_rate a todas as torres ativas.

        Chamada pelo main.py logo após posicionar a torre no campo.
        Não faz nada se o buff já estiver ativo (anti-stack).
        """
        ja_ativo = any(
            isinstance(t, DrivingCarSpeed) and t is not self and t.buff_aplicado
            for t in towers
        )
        if ja_ativo:
            logger.info("DrivingCarSpeed: buff já ativo — segunda instância ignorada.")
            return

        self.buff_aplicado = True
        for t in towers:
            if t is not self and t.fire_rate > 0:
                t.fire_rate *= BUFF_FIRE_RATE_MULT
        logger.info("DrivingCarSpeed: buff de cadência aplicado a %d torres.", len(towers))

    def on_remove(self, towers: list["Tower"]) -> None:
        """Reverte o buff de fire_rate ao remover a torre do campo.

        Chamada pelo main.py antes de remover a torre da lista.
        """
        if not self.buff_aplicado:
            return

        self.buff_aplicado = False
        for t in towers:
            if t is not self and t.fire_rate > 0:
                t.fire_rate /= BUFF_FIRE_RATE_MULT
        logger.info("DrivingCarSpeed: buff revertido.")

    def draw_range(self, surface) -> None:
        """Sem círculo de alcance em combate — range é só para exibição."""
        return
```

def rolar(n: int = 1) -> dict:
    """Realiza `n` rolls do gacha com sistema de pity.

    Retorna:
      - {"erro": "saldo_insuficiente", "saldo": int}
      - {"gasto": int, "ganhos": list[str], "saldo": int, "pity_counter": int}
    """
    custo = PRECO_1X * n
    dados = _carregar()
    if dados["saldo"] < custo:
        return {"erro": "saldo_insuficiente", "saldo": dados["saldo"]}

    dados["saldo"] -= custo
    ganhos: list[str] = []

    for _ in range(n):
        dados["pity_counter"] += 1
        item = None

        # Pity: ao atingir o limite, garante DrivingCarSpeed
        pity_garantido = dados["pity_counter"] >= PITY_LIMITE

        if pity_garantido or random.random() < CHANCE_GANHAR:
            item = ITEM_DRIVING_CAR_SPEED
        elif random.random() < CHANCE_GANHAR:
            item = PLACEHOLDER_ITEM

        if item is not None:
            ganhos.append(item)
            if item not in dados["itens"]:
                dados["itens"].append(item)
            # Resetar pity ao obter DrivingCarSpeed
            if item == ITEM_DRIVING_CAR_SPEED:
                dados["pity_counter"] = 0

    _salvar(dados)
    return {
        "gasto": custo,
        "ganhos": ganhos,
        "saldo": dados["saldo"],
        "pity_counter": dados["pity_counter"],
    }
```

                self._sprites[tipo.asset_name] = pygame.transform.smoothscale(
                    assets.get(tipo.asset_name), (SPRITE_SZ, SPRITE_SZ)
                )
            except KeyError:
                pass

        self._resultado: str | None = None   # 'fechar' quando deve sair

    # ------------------------------------------------------------------ #
    def _calcular_slots(self, painel: pygame.Rect) -> list[pygame.Rect]:
        """5 slots de deck dispostos horizontalmente no painel."""
        total_w = DECK_SLOTS * CARD_W + (DECK_SLOTS - 1) * CARD_GAP
        x0 = painel.x + (painel.width - total_w) // 2
        y0 = painel.y + 80
        return [
            pygame.Rect(x0 + i * (CARD_W + CARD_GAP), y0, CARD_W, CARD_H)
            for i in range(DECK_SLOTS)
        ]

    def _coleção_disponiveis(self) -> list[type]:
        """Tipos de torres disponíveis (todas base + gacha obtidas)."""
        itens_gacha = texas_coins.get_itens()
        disponiveis = []
        for tipo in TOWER_TYPES:
            # Cartas gacha: só aparecem se obtidas
            from entities.driving_car_speed import DrivingCarSpeed
            if tipo is DrivingCarSpeed:
                if texas_coins.ITEM_DRIVING_CAR_SPEED not in itens_gacha:
                    continue
            disponiveis.append(tipo)
        return disponiveis

    # ------------------------------------------------------------------ #
    # Input
    # ------------------------------------------------------------------ #
    def handle_event(self, event: pygame.event.Event) -> str | None:
        """Retorna 'fechar' para sair, None para continuar."""
        if event.type == pygame.KEYDOWN and event.key == pygame.K_ESCAPE:
            self._salvar_e_fechar()
            return "fechar"

        if event.type == pygame.MOUSEBUTTONDOWN and event.button == 1:
            pos = event.pos

            # Clique em slot do deck
            for i, rect in enumerate(self._slot_rects):
                if rect.collidepoint(pos):
                    if i < len(self._deck):
                        # Desequipa a torre
                        self._deck.pop(i)
                        self._slot_selecionado = None
                    else:
                        self._slot_selecionado = i
                    return None

            # Clique em carta da coleção
            tipos = self._coleção_disponiveis()
            for i, rect in enumerate(self._colecao_rects):
                if rect.collidepoint(pos) and i < len(tipos):
                    tipo = tipos[i]
                    if tipo.asset_name not in self._deck and len(self._deck) < DECK_SLOTS:
                        self._deck.append(tipo.asset_name)
                        self._slot_selecionado = None
                    return None

            # Clique fora dos painéis → fecha
            if not self._painel_deck.collidepoint(pos) and not self._painel_colecao.collidepoint(pos):
                self._salvar_e_fechar()
                return "fechar"

        return None

    def _salvar_e_fechar(self) -> None:
        texas_coins.salvar_deck(self._deck)

    # ------------------------------------------------------------------ #
    # Render
    # ------------------------------------------------------------------ #
    def draw(self, surface: pygame.Surface) -> None:
        surface.fill(COR_FUNDO_TELA)

        titulo = self._fonte_big.render("GERENCIAR DECK", True, COLOR_GOLD)
        surface.blit(titulo, titulo.get_rect(center=(WINDOW_WIDTH // 2, 52)))

        self._draw_painel(surface, self._painel_deck, "DECK ATIVO")
        self._draw_painel(surface, self._painel_colecao, "COLEÇÃO")

        self._draw_slots(surface)
        self._draw_colecao(surface)

        hint = self._fonte_label.render(
            "Clique na coleção para equipar  |  Clique no deck para desequipar  |  ESC: fechar",
            True, COR_LABEL_HUD,
        )
        surface.blit(hint, hint.get_rect(center=(WINDOW_WIDTH // 2, WINDOW_HEIGHT - 28)))

    def _draw_painel(self, surface: pygame.Surface, rect: pygame.Rect, titulo: str) -> None:
        pygame.draw.rect(surface, COR_FUNDO_MODAL, rect)
        pygame.draw.rect(surface, COR_BORDA_MODAL, rect, 1)
        pygame.draw.line(surface, COLOR_GOLD, rect.topleft, (rect.right, rect.top), 3)
        hdr = self._fonte_hdr.render(titulo, True, COR_DOURADO)
        surface.blit(hdr, (rect.x + 16, rect.y + 14))

    def _draw_slots(self, surface: pygame.Surface) -> None:
        for i, rect in enumerate(self._slot_rects):
            equipada = self._deck[i] if i < len(self._deck) else None
            cor_borda = COLOR_GOLD if i == self._slot_selecionado else COR_BORDA_MODAL
            pygame.draw.rect(surface, (16, 14, 3), rect)
            pygame.draw.rect(surface, cor_borda, rect, 2)

            if equipada and equipada in self._sprites:
                sprite = self._sprites[equipada]
                surface.blit(sprite, sprite.get_rect(center=(rect.centerx, rect.centery - 10)))
                tipo = next((t for t in TOWER_TYPES if t.asset_name == equipada), None)
                if tipo:
                    nome = self._fonte_label.render(tipo.nome[:12], True, COR_TEXTO)
                    surface.blit(nome, nome.get_rect(centerx=rect.centerx, y=rect.bottom - 20))
            else:
                vazio = self._fonte_label.render(f"Slot {i+1}", True, COR_LABEL_HUD)
                surface.blit(vazio, vazio.get_rect(center=rect.center))

    def _draw_colecao(self, surface: pygame.Surface) -> None:
        tipos = self._coleção_disponiveis()
        p = self._painel_colecao
        cols = (p.width - 32) // (CARD_W + CARD_GAP)
        self._colecao_rects = []

        for i, tipo in enumerate(tipos):
            col = i % cols
            row = i // cols
            x = p.x + 16 + col * (CARD_W + CARD_GAP)
            y = p.y + 60 + row * (CARD_H + CARD_GAP)
            rect = pygame.Rect(x, y, CARD_W, CARD_H)
            self._colecao_rects.append(rect)

            equipada = tipo.asset_name in self._deck
            cor_borda = COR_VERDE_NEON if equipada else COR_BORDA_MODAL
            pygame.draw.rect(surface, (16, 14, 3), rect)
            pygame.draw.rect(surface, cor_borda, rect, 2 if equipada else 1)

            if tipo.asset_name in self._sprites:
                sprite = self._sprites[tipo.asset_name]
                surface.blit(sprite, sprite.get_rect(center=(rect.centerx, rect.centery - 12)))

            nome_surf = self._fonte_label.render(tipo.nome[:12], True, COR_TEXTO)
            surface.blit(nome_surf, nome_surf.get_rect(centerx=rect.centerx, y=rect.bottom - 22))

            custo_surf = self._fonte_label.render(f"${tipo.cost}", True, COLOR_GOLD)
            surface.blit(custo_surf, (rect.x + 4, rect.y + 4))

    def destroy(self) -> None:
        self._salvar_e_fechar()
```

def registrar_vitoria(
    nome: str,
    tempo_segundos: float,
    category: str = "normal_speedrun",
) -> int | None:
    """Registra vitória com categoria. Retrocompatível com callers sem category."""
    player_id = _get_or_create_player_id()
    nome_sanitizado = nome[:16].strip() or "Anônimo"
    novo_tempo = round(tempo_segundos, 1)

    existente = _request(
        "GET",
        f"{TABELA}?player_id=eq.{player_id}&category=eq.{category}&select=tempo",
    )
    if existente:
        tempo_salvo = existente[0].get("tempo", float("inf"))
        if novo_tempo >= tempo_salvo:
            top = buscar_top10(category)
            for i, e in enumerate(top):
                if e.get("player_id") == player_id:
                    return i + 1
            return None

    entrada = {
        "nome": nome_sanitizado,
        "tempo": novo_tempo,
        "valor": novo_tempo,
        "player_id": player_id,
        "category": category,
    }
    _request("POST", f"{TABELA}?on_conflict=player_id,category", entrada, upsert=True)

    top = buscar_top10(category)
    for i, e in enumerate(top):
        if e.get("player_id") == player_id:
            return i + 1
    return None
```
